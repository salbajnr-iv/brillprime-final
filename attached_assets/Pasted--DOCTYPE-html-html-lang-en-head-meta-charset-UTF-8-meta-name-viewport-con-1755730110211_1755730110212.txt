    <script>
        // Add interactivity
        function goBack() {
            console.log('Going back...');
            // Add haptic feedback simulation
            if (navigator.vibrate) {
                navigator.vibrate(50);
            }
            // You could add navigation logic here
        }

        function setLocationAutomatically() {
            console.log('Setting location automatically...');
            
            // Simulate loading state
            const btn = document.querySelector('.set-automatically-btn');
            const originalText = btn.textContent;
            btn.textContent = 'Getting location...';
            btn.style.opacity = '0.7';
            
            // Simulate geolocation request
            if (navigator.geolocation) {
                navigator.geolocation.getCurrentPosition(
                    function(position) {
                        console.log('Location obtained:', position.coords);
                        btn.textContent = 'Location set!';
                        btn.style.background = '#28a745';
                        
                        setTimeout(() => {
                            btn.textContent = originalText;
                            btn.style.opacity = '1';
                            btn.style.background = '#4682B4';
                        }, 2000);
                    },
                    function(error) {
                        console.log('Location error:', error);
                        btn.textContent = 'Location access denied';
                        btn.style.background = '#dc3545';
                        
                        setTimeout(() => {
                            btn.textContent = originalText;
                            btn.style.opacity = '1';
                            btn.style.background = '#4682B4';
                        }, 2000);
                    }
                );
            } else {
                btn.textContent = 'Geolocation not supported';
                setTimeout(() => {
                    btn.textContent = originalText;
                    btn.style.opacity = '1';
                }, 2000);
            }
        }

        function setLocationLater() {
            console.log('Setting location later...');
            
            const btn = document.querySelector('.set-later-btn');
            const originalText = btn.textContent;
            btn.textContent = 'Skipped';
            btn.style.background = '#6c757d';
            btn.style.color = 'white';
            btn.style.borderColor = '#6c757d';
            
            setTimeout(() => {
                btn.textContent = originalText;
                btn.style.background = 'transparent';
                btn.style.color = '#131313';
                btn.style.borderColor = '#4682B4';
            }, 1500);
        }

        // Add some interactive animations
        document.addEventListener('DOMContentLoaded', function() {
            const modal = document.querySelector('.location-modal');
            const icon = document.querySelector('.location-icon');
            
            // Animate modal entrance
            setTimeout(() => {
                modal.style.transform = 'translateY(-10px)';
                modal.style.transition = 'transform 0.6s ease-out';
                
                setTimeout(() => {
                    modal.style.transform = 'translateY(0)';
                }, 100);
            }, 500);
            
            // Pulse animation for location icon
            setInterval(() => {
                icon.style.transform = 'scale(1.05)';
                setTimeout(() => {
                    icon.style.transform = 'scale(1)';
                }, 300);
            }, 3000);
        });

        // Add touch/click effects
        document.querySelectorAll('button, .back-button-circle').forEach(element => {
            element.addEventListener('touchstart', function(e) {
                this.style.transform = 'scale(0.95)';
            });
            
            element.addEventListener('touchend', function(e) {
                this.style.transform = '';
            });
            
            element.addEventListener('mousedown', function(e) {
                this.style.transform = 'scale(0.95)';
            });
            
            element.addEventListener('mouseup', function(e) {
                this.style.transform = '';
            });
        });
    </script>
